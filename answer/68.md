### 해답
```
$ f=0; t=$(bc <<< 2^100); while [ $(bc <<< $t-$f) != 1 ];do m=$(bc <<< "($f+$t)/2"); echo $m; date -d @$m && f=$m || t=$m; done
```

別解1 $ n=0; i=63; while [[ i -gt 0 ]]; do date -d @$n >& /dev/null; ((n += ($? ? -1 : 1) << --i)); done; echo $n; date -d @$n
